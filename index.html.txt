<script>
  // Prosty "licznik odwiedzin" (localStorage)
  (function(){
    try {
      var hits = localStorage.getItem('retro_hits') || 0;
      hits = Number(hits) + 1;
      localStorage.setItem('retro_hits', hits);
      document.getElementById('hits').textContent = hits;
    } catch(e) {
      document.getElementById('hits').textContent = '---';
    }
  })();

  // "Goście online" (losowa symulacja)
  (function(){
    var el = document.getElementById('visitorCount');
    function update() {
      var n = 1 + Math.floor(Math.random()*8);
      el.textContent = n + ' odwiedzających';
    }
    update();
    setInterval(update, 5000);
  })();

  // Prosta księga gości (localStorage)
  function renderEntries() {
    var container = document.getElementById('entries');
    container.innerHTML = '';
    var data = JSON.parse(localStorage.getItem('retro_guestbook') || '[]');
    if (data.length === 0) {
      container.innerHTML = '<p>Brak wpisów. Bądź pierwszy!</p>';
      return;
    }
    data.slice().reverse().forEach(function(e){
      var div = document.createElement('div');
      div.className = 'entry';
      div.innerHTML = '<strong>' + escapeHtml(e.name) + '</strong> <em>(' + e.time + ')</em><div>' + escapeHtml(e.msg) + '</div>';
      container.appendChild(div);
    });
  }

  function addEntry() {
    var name = document.getElementById('gbName').value.trim();
    var msg = document.getElementById('gbMessage').value.trim();
    if (!name || !msg) { alert('Wpisz imię i wiadomość.'); return false; }
    var data = JSON.parse(localStorage.getItem('retro_guestbook') || '[]');
    data.push({name: name, msg: msg, time: (new Date()).toLocaleString()});
    localStorage.setItem('retro_guestbook', JSON.stringify(data));
    document.getElementById('gbName').value = '';
    document.getElementById('gbMessage').value = '';
    renderEntries();
    return false;
  }

  function escapeHtml(s) {
    return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }

  // inicjalizacja
  renderEntries();
</script>
